model districtblockmap_vandan {
  id         Int      @id @default(autoincrement())
  districtId Int
  blockId    Int
  block_vd   block_vd_vandan @relation(fields: [blockId], references: [bolck_Id], onDelete: Cascade, map: "DistrictBlockMap_blockId_fkey")
  district   district_vandan @relation(fields: [districtId], references: [district_id], onDelete: Cascade, map: "DistrictBlockMap_districtId_fkey")

  @@unique([districtId, blockId], map: "Unique_District_Block")
  @@index([blockId], map: "DistrictBlockMap_blockId_idx")
  @@index([districtId], map: "DistrictBlockMap_districtId_idx")
}
