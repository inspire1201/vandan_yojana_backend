model districtblockmap_vandan {
  id         Int @id @default(autoincrement())
  districtId Int
  blockId    Int
  
  district_vandan district_vandan @relation(fields: [districtId], references: [district_id])
  block_vd_vandan block_vd_vandan @relation(fields: [blockId], references: [bolck_Id])

  @@unique([districtId, blockId], map: "Unique_District_Block")
  @@index([blockId], map: "DistrictBlockMap_blockId_idx")
  @@index([districtId], map: "DistrictBlockMap_districtId_idx")
}
